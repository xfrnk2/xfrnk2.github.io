나의 풀이
def get_result(sentence: str) -> bool:
    bracket_stack = []

    for s in sentence:

        if s.isalpha():
            continue

        elif s == "(":
            bracket_stack.append(1)

        elif s == "[":
            bracket_stack.append(0)

        elif s == ")" or s == "]":

            try:
                if bracket_stack[-1] == 0 and s == ")":
                    return False
                elif bracket_stack[-1] == 1 and s == "]":
                    return False
                bracket_stack.pop()

            except IndexError:
                return False

    return not bracket_stack

    sentence_list = []

    while True:
        get_input = input()

  • 괄호의 종류가 일치하면 pop
